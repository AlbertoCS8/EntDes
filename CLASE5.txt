SELECT OFICIO 
 FROM EMPLE
 WHERE APELLIDO = 'GIL';
 
SELECT APELLIDO , OFICIO
 FROM EMPLE
 WHERE OFICIO = 'ANALISTA'; 
 
SELECT APELLIDO
  FROM EMPLE
  WHERE DIR = (SELECT DIR
               FROM EMPLE
               WHERE APELLIDO='SALA'
               )
        AND
        APELLIDO <> 'SALA';
        
 SELECT APELLIDO
   FROM EMPLE
   WHERE EMP_NO = ( SELECT DIR
                    FROM EMPLE
                    WHERE APELLIDO = 'REY'
                    );
                        
SELECT APELLIDO, SALARIO
   FROM EMPLE
   WHERE OFICIO = (SELECT OFICIO
                   FROM EMPLE
                   WHERE APELLIDO = 'SÁNCHEZ'
                   );
                   
                                          
 SELECT APELLIDO,SALARIO
   FROM EMPLE
   WHERE SALARIO > (SELECT SALARIO  
                    FROM EMPLE
                    WHERE APELLIDO = 'CEREZO'
                    )
   ORDER BY  SALARIO DESC;

  FROM EMPLE
  WHERE FECHA_ALT = (SELECT FECHA_ALT
                     FROM EMPLE

                     )

        APELLIDO <> 'JIMENO';
