  FROM DEPART
  WHERE DEPT_NO IN ( SELECT DEPT_NO
                     FROM EMPLE

                    =

                      SELECT ROUND ( AVG (TO_CHAR(FECHA_ALT,'MM')) ) + 4 AS "MES MEDIO DE ALTA"
                       FROM EMPLE  
                     ));
