 EMPLE DEPART


SELECT APENOM, NOMBRE
   FROM ALUMNOS AL
     INNER JOIN NOTAS N
      ON AL.DNI = N.DNI
       INNER JOIN ASIGNATURAS AG
         ON AG.COD = N.COD;
         
SELECT DISTINCT APENOM
  FROM ALUMNOS AL
   INNER JOIN NOTAS N
    ON AL.DNI = N.DNI
    WHERE NOTA >= 6;
    
    
SELECT APENOM, NOTA ,NOMBRE
  FROM ALUMNOS AL
    INNER JOIN NOTAS N
      ON AL.DNI = N.DNI
       INNER JOIN ASIGNATURAS AG
        ON N.COD = AG.COD
  WHERE  NOMBRE = 'FOL';         
      
SELECT APENOM, NOTA ,NOMBRE
  FROM ALUMNOS AL
    NATURAL JOIN NOTAS N
       INNER JOIN ASIGNATURAS AG
        ON N.COD = AG.COD
  WHERE  NOMBRE = 'FOL';      
   
  
SELECT APELLIDO , LOC
  FROM EMPLE E, DEPART D
    WHERE E.DEPT_NO = D.DEPT_NO;
    
SELECT APELLIDO, LOC
  FROM EMPLE 
   NATURAL JOIN DEPART;

SELECT APELLIDO, LOC
  FROM EMPLE E
   INNER JOIN DEPART D
    ON E.DEPT_NO = D.DEPT_NO; 
    
SELECT APELLIDO, LOC
  FROM EMPLE E
   INNER JOIN DEPART D
    ON E.DEPT_NO = D.DEPT_NO
  WHERE LOC <> 'BARCELONA';  
  
  
 SELECT APELLIDO, DNOMBRE , SALARIO
   FROM EMPLE E
    INNER JOIN DEPART D
     ON E.DEPT_NO = D.DEPT_NO
   WHERE SALARIO >= 200000
   ORDER BY SALARIO DESC; 
  
  SELECT APELLIDO, DNOMBRE , SALARIO , DEPT_NO
   FROM EMPLE 
   NATURAL JOIN DEPART
   WHERE SALARIO >= 200000
   ORDER BY SALARIO DESC;
